@startuml
title 简易线上购物系统
class 用户
{
    + 姓名 :string
    + 年龄 :int
    + 用户ID:int
    + 地址: string
    + 邮箱: string
    + 余额: float
    + 我的购物车: 购物车
    + 我的订单: 订单

    + 加购物车(商品):void
    + 购买(商品):void
    + 结算购物车():void
}
 
class 商品
{
    + 名字 :string
    + 价格 :float
    + 库存 :int
    + 商品ID :int
}
together 订单相关{
    
    class 订单
    {
        + 用户ID :int
        + 商品ID :int
        + 店家ID :int
        + 订单状态 :订单状态
        + 查看物流():string
    }
    enum 订单状态
    {
        待付款
        待发货
        待签收
        已签收
    }
    class 订单管理
    {
        + 订单列表:订单List<订单>
        + 订单总数 :int
        + 移除订单(订单):bool
        + 生成订单(商品,用户,int 订单数):void
    }
}

class 店家
{
    + 上架商品列表:上架商品List<商品>
    + 店家ID:int
    + 店铺名称:string
    + 删除上架商品(商品ID):void
}
class 购物车
{
    + 购物车商品数量:int
    + 购物车容量:int
    + 购物车列表:List<商品>
    + 清空购物车():void
    + 结算购物车():void
    + 移除物品(商品):void
    + 加购物车(商品,int 数量):void
}
用户 --> 订单
用户 --> 购物车
用户 ..> 商品
订单 --> 订单状态
订单管理 ..> 订单
订单管理 ..> 商品
订单管理 ..> 用户
订单管理 o-- 订单
店家 o-- 商品
购物车 ..> 商品
购物车 o-- 商品


@enduml
