class Item
{   
    + 物品类型:ItemType
    + 物品名字:string
    + 物品数量:int

    Placeable 可放置
    Consumeable 可消耗
}

    + 消耗物品(slot:ItemSlot)
    + 装备物品(slot:ItemSlot)

    + 默认使用物品(slot:ItemSlot)
    + GetItemByIndex(index:int):Item
    + GetItemByName(name:string):Item
    + GetItemIndex(item:Item):int
}

}
